6    ;   sublist item consists of a specific job, the time this job takes, followed 
7    ;   by any jobs that this job depends on. Below are specific descriptions for

13   (defn build-list [L-a L-b] 
14     (cond (symbol? L-b) 
15       (conj L-a L-b) 
16       (empty? L-b) 
17       L-a 
18       :else (build-list (conj L-a (first L-b)) (rest L-b))))

49   ; #1 sum - takes the list of tasks as an argument then returns the sum of all 
50   ;   of the days needed for the individual jobs. 
51   (defn sum 
52     ([tasks] 
53       (sum tasks 0 0)) 
54     ([tasks days i] 
55       (if (= i (count tasks)) 
56         days 
57         (sum tasks (+ (second (nth tasks i)) days) (inc i)))))

59   ; #2 predecessors - takes a specific job and the list of tasks then returns 
60   ;   a list of the immediate predecessors for that job. 
61   ;   (predecessors 'vapor_barrier_insulation tasks) should return 
62   ;   (roof install_windows_doors) 
63   (defn predecessors 
64     ([job tasks] 
65       (if (nil? job) 
66         '() 
67         (predecessors job tasks 0))) 
68     ([job tasks i] 
69       (cond (= i (count tasks)) 
70         '() 
71         (= job (first (nth tasks i))) 
72         (rest (rest (nth tasks i))) 
73         :else (predecessors job tasks (inc i))))) 
74    
75   ; #3 gettime - takes a job and the list of tasks then returns the time that 
76   ;   job takes 
77   (defn gettime 
78     ([job tasks] 
79       (if (nil? job) 
80         nil 
81         (gettime job tasks 0))) 
82     ([job tasks i] 
83       (cond (= i (count tasks)) 
84         nil 
85         (= job (first (nth tasks i))) 
86         (second (nth tasks i)) 
87         :else (gettime job tasks (inc i)))))

89   ; conj item x to list L only if item doesn't currently exist in list 
90   (defn conj-no-dupe 
91     ([L x] 
92       (conj-no-dupe L x 0)) 
93     ([L x i] 
94       (cond (= i (count L)) 
95         (conj L x) 
96         (= (nth L i) x) 
97         L 
98         :else (conj-no-dupe L x (inc i)))))

100  ; #4 get-all-preds - takes a specific job and the list of tasks then returns 
101  ;   a list of all of the predecessors for that job 
102  (defn get-all-preds 
103    ([job tasks] 
104      (get-all-preds job (predecessors job tasks) tasks '())) 
105    ([job preds tasks all-preds] 
106      (if (empty? preds) 
107        all-preds 
108        (get-all-preds (first preds) 
109          (build-list 
110            (predecessors 
111              (first preds) tasks) (rest preds)) 
112          tasks (conj-no-dupe all-preds (first preds))))))

114  ; #5 precedes - takes two specific jobs and the list of tasks then returns 
115  ;   true if the first job must precede the other and nil otherwise 
116  ; check to see if item exitsts in list 
117  (defn not-exists? 
118    ([L x] 
119      (not-exists? L x 0)) 
120    ([L x i] 
121      (cond (= i (count L)) 
122        true 
123        (= (nth L i) x) 
124        false 
125        :else (not-exists? L x (inc i))))) 
126   
127  (defn precedes [J1 J2 L] 
128    (if (not-exists? (get-all-preds J1 tasks) J2) 
129      true 
130      nil))

132  ; #6 start_day - takes a specific job and the list of tasks then returns the 
133  ;   day that this job can start 
134  (defn start-day [job tasks] 
135    (if (empty? (predecessors job tasks)) 
136      1 
137      (first (get-max-help (predecessors job tasks) tasks))))

139  ; #7 get_max - takes a list of job names and the list of tasks then returns a 
140  ;   list with the time and the job that finishes at the greatest time 
141  ;   ==> (get_max '(frame roof select_subs) tasks) 
142  ;   should return: 
143  ;   (37 roof) ;since roof cannot finish until day 37 
144  ;   (assuming the entire project started on day 1). 
145  (defn get-max [preds tasks] 
146    (cons (dec (first (get-max-help preds tasks))) (rest (get-max-help preds tasks)))) 
147   
148  (defn get-max-help [preds tasks] 
149    (cond (empty? preds) 
150      '() 
151      (= 1 (count preds)) 
152      (list (+ (start-day (first preds) tasks) 
153              (gettime (first preds) tasks)) (first preds)) 
154      (> (+ (start-day (first preds) tasks) 
155           (gettime (first preds) tasks)) (first (get-max-help (rest preds) tasks))) 
156      (list (+ (start-day (first preds) tasks) 
157              (gettime (first preds) tasks)) (first preds)) 
158      :else (get-max-help (rest preds) tasks)))

160  ; #8 critical_path - takes a job and the list of tasks then returns a list of 
161  ;   the jobs on the critical path to getting this job done in the least 
162  ;   amount of time 
163  (defn critical-path [job tasks] 
164    (if (not (symbol? job)) 
165      '() 
166      (cons (second (get-max-help (predecessors job tasks) tasks)) 
167        (critical-path 
168          (second (get-max-help (predecessors job tasks) tasks)) tasks))))

170  ; #9 depends_on - takes a job and the list of tasks then returns a list of the 
171  ;   jobs that cannot be started until this job has completed. This should 
172  ;   return all future jobs, not just the ones immediately affected. 
173  (defn depends-on 
174    ([job tasks] 
175      (depends-on job tasks '())) 
176    ([job tasks dependents] 
177      (cond (empty? tasks) 
178        dependents 
179        (= job (first (first tasks))) 
180        (depends-on job (rest tasks) dependents) 
181        (precedes job (first (first tasks)) tasks) 
182        (depends-on job 
183          (rest tasks) 
184          (build-list dependents 
185            (first (first tasks)))) 
186        :else (depends-on job (rest tasks) dependents))))
